/**
 * @param {number[]} nums
 * @param {number} val
 * @return {number}
 * 思路：快慢指针
 */
 var removeElement = function(nums, val) {
    if (nums.length == 0) {
        return 0;
    }
    let slow = 0, fast = 0;
    while (fast < nums.length) {
        if (nums[fast] !== val) {
            nums[slow] = nums[fast];
            ++slow;
        }
        ++fast;
    }
    return slow;
};
